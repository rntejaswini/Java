export default function Certificates() {
  const [selected, setSelected] = useState(null);

  // Convert imported images into array
  const certificates = Object.values(images).map((img) => {
    const filePath = img.default;
    const fileName = filePath
      .split("/")
      .pop()
      .replace(/\.(png|jpg|jpeg)/i, "")
      .replace(/%20/g, " ")
      .replace(/[-_]/g, " ")
      .trim();

    return {
      image: filePath,
      title: fileName,
    };
  });

  // Download function
  const downloadCertificate = (image, title) => {
    const link = document.createElement("a");
    link.href = image;
    link.download = `${title}.png`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  return (
  <section id="certificates" className="py-16 bg-white dark:bg-gray-950">

      <div className="max-w-7xl mx-auto px-6">
        
        {/* Heading */}
        <h2 className="text-3xl font-bold text-center text-white mb-10">
          Certifications
        </h2>

        {/* Certificates Grid */}
        <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
          {certificates.map((cert, index) => (
            <div
              key={index}
              onClick={() => setSelected(cert)}
              className="bg-gray-900 rounded-xl overflow-hidden shadow-lg cursor-pointer hover:scale-105 transition-transform"
            >
              <img
                src={cert.image}
                alt={cert.title}
                className="w-full h-48 object-cover"
              />
              <div className="p-3 text-center">
                <p className="text-sm font-semibold text-white">
                  {cert.title}
                </p>
              </div>
            </div>
          ))}
        </div>

        {/* Modal */}
        {selected && (
          <div className="fixed inset-0 bg-black bg-opacity-80 flex items-center justify-center z-50">
            <div className="relative max-w-3xl w-full p-4">

              {/* Close Button */}
              <button
                onClick={() => setSelected(null)}
                className="absolute top-2 right-2 text-white text-2xl"
              >
                ‚úï
              </button>

              {/* Certificate Image */}
              <img
                src={selected.image}
                alt={selected.title}
                className="w-full rounded-lg"
              />

              {/* Title */}
              <p className="text-center text-white mt-4 font-semibold">
                {selected.title}
              </p>

              {/* Download Button */}
              <div className="flex justify-center mt-4">
                <button
                  onClick={() =>
                    downloadCertificate(selected.image, selected.title)
                  }
                  className="bg-blue-600 hover:bg-blue-700 text-white px-6 py-2 rounded-lg font-semibold transition"
                >
                  ‚¨áÔ∏è Download Certificate
                </button>
              </div>

            </div>
          </div>
        )}
      </div>
    </section>
  );

export default function Navbar() {
  const [theme, setTheme] = useState("light");

  useEffect(() => {
    const savedTheme = localStorage.getItem("theme") || "light";
    setTheme(savedTheme);
    if (savedTheme === "dark") {
      document.documentElement.classList.add("dark");
    } else {
      document.documentElement.classList.remove("dark");
    }
  }, []);

  const toggleTheme = () => {
    const newTheme = theme === "light" ? "dark" : "light";
    setTheme(newTheme);
    localStorage.setItem("theme", newTheme);

    if (newTheme === "dark") {
      document.documentElement.classList.add("dark");
    } else {
      document.documentElement.classList.remove("dark");
    }
  };

  return (
    <nav className="fixed w-full bg-black/80 backdrop-blur z-50">
      <div className="max-w-6xl mx-auto px-6 py-4 flex justify-between items-center">
        <h1 className="font-bold text-xl text-white">Tejaswini RN</h1>

            Resume
          </a>

            onClick={toggleTheme}
            className="p-2 rounded-full bg-gray-200 dark:bg-gray-700 text-gray-900 dark:text-white hover:bg-gray-300 dark:hover:bg-gray-600"
