import java.util.*;
import java.io.*;

class Codechef {
    public static void main (String[] args) throws Exception {

        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();

        while (T-- > 0) {
            int N = sc.nextInt();

            double value = Math.pow(0.143 * N, N);

            long answer = Math.round(value);
            System.out.println(answer);
        }

        sc.close();
    }
}
