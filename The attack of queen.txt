import java.util.*;
import java.io.*;

class Codechef {
    public static void main(String[] args) throws Exception {
        FastScanner fs = new FastScanner(System.in);
        int T = fs.nextInt();
        StringBuilder sb = new StringBuilder();
        
        while (T-- > 0) {
            long N = fs.nextLong();
            long X = fs.nextLong();
            long Y = fs.nextLong();

            long horizontal = N - 1;
            long vertical = N - 1;

            long diagonal1 = Math.min(X - 1, Y - 1);
            long diagonal2 = Math.min(N - X, N - Y);
            long diagonal3 = Math.min(X - 1, N - Y);
            long diagonal4 = Math.min(N - X, Y - 1);

            long result = horizontal + vertical + diagonal1 + diagonal2 + diagonal3 + diagonal4;
            sb.append(result).append("\n");
        }

        System.out.print(sb.toString());
    }

    // Fast input class
    static class FastScanner {
        private final InputStream in;
        private final byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        FastScanner(InputStream is) {
            in = is;
        }

        private int read() throws IOException {
            if (ptr >= len) {
                len = in.read(buffer);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        long nextLong() throws IOException {
            long sign = 1;
            int c;
            do {
                c = read();
            } while (c <= ' ');

            if (c == '-') {
                sign = -1;
                c = read();
            }

            long val = 0;
            while (c > ' ') {
                val = val * 10 + (c - '0');
                c = read();
            }
            return val * sign;
        }

        int nextInt() throws IOException {
            return (int) nextLong();
        }
    }
}
