import java.util.*;

class Solution {
    public int mostBooked(int n, int[][] meetings) {

        Arrays.sort(meetings, (a, b) -> Integer.compare(a[0], b[0]));

        // min-heap of available rooms (by index)
        PriorityQueue<Integer> freeRooms = new PriorityQueue<>();
        for (int i = 0; i < n; i++) freeRooms.offer(i);

        // min-heap of busy rooms: [endTime, roomIndex]
        PriorityQueue<long[]> busy = new PriorityQueue<>(
                (a, b) -> a[0] == b[0] ? Long.compare(a[1], b[1])
                                       : Long.compare(a[0], b[0])
        );

        int[] count = new int[n];

        for (int[] m : meetings) {
            long start = m[0], end = m[1];
            long duration = end - start;

            // free rooms whose meetings are finished
            while (!busy.isEmpty() && busy.peek()[0] <= start) {
                freeRooms.offer((int) busy.poll()[1]);
            }

            int room;
            long actualStart, actualEnd;

            if (!freeRooms.isEmpty()) {
                // schedule immediately
                room = freeRooms.poll();
                actualStart = start;
                actualEnd = start + duration;
            } else {
                // delay meeting until earliest room frees up
                long[] top = busy.poll();
                actualStart = top[0];
                room = (int) top[1];
                actualEnd = actualStart + duration;
            }

            count[room]++;
            busy.offer(new long[]{actualEnd, room});
        }

        // find room with max meetings (smallest index on tie)
        int best = 0;
        for (int i = 1; i < n; i++) {
            if (count[i] > count[best]) best = i;
        }

        return best;
    }
}
