class Result {

    public static int beautifulBinaryString(String b) {
        int count = 0;

        for (int i = 0; i <= b.length() - 3; i++) {
            if (b.substring(i, i + 3).equals("010")) {
                count++;
                i += 2; // skip ahead to avoid overlap
            }
        }

        return count;
    }
}
