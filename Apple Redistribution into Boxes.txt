import java.util.Arrays;

class Solution {
    public int minimumBoxes(int[] apple, int[] capacity) {
        // Find total apples
        int totalApples = 0;
        for (int a : apple) {
            totalApples += a;
        }

        // Sort capacities in descending order
        Arrays.sort(capacity);
        int usedBoxes = 0;
        int currCapacity = 0;

        // Pick largest boxes first
        for (int i = capacity.length - 1; i >= 0; i--) {
            currCapacity += capacity[i];
            usedBoxes++;
            if (currCapacity >= totalApples) {
                break;
            }
        }

        return usedBoxes;
    }
}
