#include <stdio.h>

int digitSum(int h, int m) {
    return (h / 10) + (h % 10) + (m / 10) + (m % 10);
}

int main() {
    int K;
    char time[6];

    scanf("%s", time);
    scanf("%d", &K);

    int hour = (time[0] - '0') * 10 + (time[1] - '0');
    int min  = (time[3] - '0') * 10 + (time[4] - '0');

    for (int wait = 0; wait < 1440; wait++) {

        if (digitSum(hour, min) % K == 0) {
            printf("%d", wait);
            return 0;
        }

        // Move to next minute
        min++;
        if (min == 60) {
            min = 0;
            hour++;
            if (hour == 24) {
                hour = 0;
            }
        }
    }

    // If no good time found in 24 hours
    printf("-1");
    return 0;
}
