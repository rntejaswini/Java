import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

class TestClass {

    static int targetV, targetC, targetF, targetP;
    static int n;
    static int[][] fruits;
    static boolean possible = false;

    static void dfs(int idx, int v, int c, int f, int p) {
        // If already found a valid subset, stop further recursion
        if (possible) return;

        // If sums exceed targets, prune
        if (v > targetV || c > targetC || f > targetF || p > targetP)
            return;

        // If all targets matched exactly
        if (v == targetV && c == targetC && f == targetF && p == targetP) {
            possible = true;
            return;
        }

        // If all fruits processed
        if (idx == n) return;

        // Option 1: include current fruit
        dfs(idx + 1,
            v + fruits[idx][0],
            c + fruits[idx][1],
            f + fruits[idx][2],
            p + fruits[idx][3]);

        // Option 2: exclude current fruit
        dfs(idx + 1, v, c, f, p);
    }

    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        targetV = Integer.parseInt(st.nextToken());
        targetC = Integer.parseInt(st.nextToken());
        targetF = Integer.parseInt(st.nextToken());
        targetP = Integer.parseInt(st.nextToken());

        n = Integer.parseInt(br.readLine());
        fruits = new int[n][4];

        for (int i = 0; i < n; i++) {
            st = new StringTokenizer(br.readLine());
            fruits[i][0] = Integer.parseInt(st.nextToken());
            fruits[i][1] = Integer.parseInt(st.nextToken());
            fruits[i][2] = Integer.parseInt(st.nextToken());
            fruits[i][3] = Integer.parseInt(st.nextToken());
        }

        dfs(0, 0, 0, 0, 0);

        System.out.println(possible ? "YES" : "NO");
    }
}
