import java.io.*;
import java.util.*;

class Result {

    public static List<Integer> missingNumbers(List<Integer> arr, List<Integer> brr) {

        Map<Integer, Integer> freqArr = new HashMap<Integer, Integer>();
        Map<Integer, Integer> freqBrr = new HashMap<Integer, Integer>();

        // Count frequency in arr
        for (int num : arr) {
            if (freqArr.containsKey(num)) {
                freqArr.put(num, freqArr.get(num) + 1);
            } else {
                freqArr.put(num, 1);
            }
        }

        // Count frequency in brr
        for (int num : brr) {
            if (freqBrr.containsKey(num)) {
                freqBrr.put(num, freqBrr.get(num) + 1);
            } else {
                freqBrr.put(num, 1);
            }
        }

        List<Integer> result = new ArrayList<Integer>();

        // Find missing numbers
        for (int key : freqBrr.keySet()) {
            int countInBrr = freqBrr.get(key);
            int countInArr = freqArr.containsKey(key) ? freqArr.get(key) : 0;

            if (countInBrr > countInArr) {
                result.add(key);
            }
        }

        Collections.sort(result);
        return result;
    }
}

public class Solution {
    public static void main(String[] args) throws IOException {

        BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bufferedWriter = new BufferedWriter(new OutputStreamWriter(System.out));

        int n = Integer.parseInt(bufferedReader.readLine().trim());
        String[] arrItems = bufferedReader.readLine().trim().split(" ");
        List<Integer> arr = new ArrayList<Integer>();
        for (String item : arrItems) {
            arr.add(Integer.parseInt(item));
        }

        int m = Integer.parseInt(bufferedReader.readLine().trim());
        String[] brrItems = bufferedReader.readLine().trim().split(" ");
        List<Integer> brr = new ArrayList<Integer>();
        for (String item : brrItems) {
            brr.add(Integer.parseInt(item));
        }

        List<Integer> result = Result.missingNumbers(arr, brr);

        for (int i = 0; i < result.size(); i++) {
            bufferedWriter.write(String.valueOf(result.get(i)));
            if (i != result.size() - 1) {
                bufferedWriter.write(" ");
            }
        }

        bufferedWriter.newLine();
        bufferedReader.close();
        bufferedWriter.close();
    }
}
