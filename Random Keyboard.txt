import java.io.*;
import java.util.*;

public class TestClass {
    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter wr = new PrintWriter(System.out);

        int N = Integer.parseInt(br.readLine().trim());
        String[] keyboard = new String[N];

        for (int i = 0; i < N; i++) {
            keyboard[i] = br.readLine().trim();
        }

        String word = br.readLine().trim();

        int out_ = solve(N, keyboard, word);
        System.out.println(out_);

        wr.close();
        br.close();
    }

    static int solve(int N, String[] keyboard, String word) {

        // Convert each keyboard row to a set of characters
        List<Set<Character>> rows = new ArrayList<>();

        for (int i = 0; i < N; i++) {
            Set<Character> set = new HashSet<>();
            for (char c : keyboard[i].toCharArray()) {
                set.add(c);
            }
            rows.add(set);
        }

        // find which row contains first character of word
        char first = word.charAt(0);
        Set<Character> targetRow = null;

        for (Set<Character> row : rows) {
            if (row.contains(first)) {
                targetRow = row;
                break;
            }
        }

        // if first character not found in any row, return 0
        if (targetRow == null) return 0;

        // check whether all characters belong to the same row
        for (char c : word.toCharArray()) {
            if (!targetRow.contains(c)) {
                return 0;
            }
        }

        return 1;
    }
}
