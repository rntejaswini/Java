import java.util.*;
import java.lang.*;
import java.io.*;

class Codechef
{
    public static void main (String[] args) throws java.lang.Exception
    {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        
        while(T-- > 0){
            long N = sc.nextLong();
            
            long odds = (N + 1) / 2;
            long evens = N / 2;
            
            long result = 2 * odds * evens;
            System.out.println(result);
        }
        
        sc.close();
    }
}
