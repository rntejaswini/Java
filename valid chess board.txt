import java.io.*;
import java.util.*;

class TestClass {

    static boolean isValid(char[][] grid, int r, int c, char start) {
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 8; j++) {
                char expected = ((i + j) % 2 == 0) ? start : (start == '0' ? '1' : '0');
                if (grid[r + i][c + j] != expected)
                    return false;
            }
        }
        return true;
    }

    public static void main(String args[]) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        String[] first = br.readLine().split(" ");
        int n = Integer.parseInt(first[0]);
        int m = Integer.parseInt(first[1]);

        char[][] grid = new char[n][m];
        for (int i = 0; i < n; i++) {
            grid[i] = br.readLine().toCharArray();
        }

        int count = 0;

        for (int i = 0; i + 7 < n; i++) {
            for (int j = 0; j + 7 < m; j++) {
                if (isValid(grid, i, j, '0') || isValid(grid, i, j, '1')) {
                    count++;
                }
            }
        }

        System.out.println(count);
    }
}
