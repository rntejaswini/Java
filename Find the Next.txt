import java.io.*;
import java.util.*;

class TestClass {
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        String[] first = br.readLine().trim().split(" ");
        int N = Integer.parseInt(first[0]);
        int Q = Integer.parseInt(first[1]);

        long[] arr = new long[N];
        String[] input = br.readLine().trim().split(" ");

        for (int i = 0; i < N; i++) {
            arr[i] = Long.parseLong(input[i]);
        }

        // Sort and remove duplicates
        Arrays.sort(arr);
        ArrayList<Long> unique = new ArrayList<>();
        for (long num : arr) {
            if (unique.isEmpty() || unique.get(unique.size() - 1) != num) {
                unique.add(num);
            }
        }

        // Build consecutive intervals
        ArrayList<long[]> intervals = new ArrayList<>();
        long start = unique.get(0);
        long end = unique.get(0);

        for (int i = 1; i < unique.size(); i++) {
            long num = unique.get(i);
            if (num == end + 1) {
                end = num;
            } else {
                intervals.add(new long[]{start, end});
                start = num;
                end = num;
            }
        }
        intervals.add(new long[]{start, end});

        StringBuilder sb = new StringBuilder();

        while (Q-- > 0) {
            long x = Long.parseLong(br.readLine().trim());
            long candidate = x + 1;

            // Binary search to find interval that may contain candidate
            int low = 0, high = intervals.size() - 1;
            boolean found = false;

            while (low <= high) {
                int mid = (low + high) / 2;
                long[] interval = intervals.get(mid);

                if (candidate < interval[0]) {
                    high = mid - 1;
                } else if (candidate > interval[1]) {
                    low = mid + 1;
                } else {
                    // candidate is inside interval
                    sb.append(interval[1] + 1).append("\n");
                    found = true;
                    break;
                }
            }

            if (!found) {
                sb.append(candidate).append("\n");
            }
        }

        System.out.print(sb.toString());
    }
}
