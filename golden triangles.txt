import java.util.Scanner;

class TestClass {
    public static void main(String args[]) throws Exception {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();   // number of rectangles
        int count = 0;

        for (int i = 0; i < n; i++) {
            int w = sc.nextInt();
            int h = sc.nextInt();

            // find ratio of longer side to shorter side
            double ratio = (double) Math.max(w, h) / Math.min(w, h);

            // check golden rectangle condition
            if (ratio >= 1.6 && ratio <= 1.7) {
                count++;
            }
        }

        System.out.println(count);
        sc.close();
    }
}
