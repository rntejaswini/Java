import java.io.*;
import java.util.*;

public class TestClass {

    public static void main(String[] args) throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int N = Integer.parseInt(br.readLine().trim());
        String S = br.readLine().trim();

        String[] comments = new String[N];

        // Read remaining comments (may be on one line or multiple lines)
        int index = 0;
        while (index < N) {
            String line = br.readLine();
            if (line == null || line.trim().isEmpty()) continue;

            String[] parts = line.split(" ");
            for (String p : parts) {
                if (index < N) {
                    comments[index++] = p;
                }
            }
        }

        System.out.println(solution(N, S, comments));
    }

    static int solution(int N, String S, String[] comments) {

        int count = 0;
        String key = S.toLowerCase();

        for (int i = 0; i < N; i++) {
            if (comments[i] == null) continue;

            String[] words = comments[i].toLowerCase().split("_");

            for (String word : words) {
                if (word.equals(key)) {
                    count++;
                    break; // count comment only once
                }
            }
        }

        return count;
    }
}
