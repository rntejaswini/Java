#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    char s[100005];
    scanf("%s", s);

    int freq[26] = {0};

    // Count frequency of each character
    for (int i = 0; i < N; i++) {
        freq[s[i] - 'a']++;
    }

    // Find maximum frequency
    int maxFreq = 0;
    for (int i = 0; i < 26; i++) {
        if (freq[i] > maxFreq) {
            maxFreq = freq[i];
        }
    }

    // Print minimum possible length
    printf("%d", N - maxFreq);

    return 0;
}
