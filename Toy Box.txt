import java.util.*;

class TestClass {
    public static void main(String args[]) throws Exception {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(); // number of toys
        int m = sc.nextInt(); // number of boxes

        int[] maxPrice = new int[m + 1]; // 1-based indexing for boxes

        for (int i = 0; i < n; i++) {
            int price = sc.nextInt();
            int box = sc.nextInt();
            // Store the max price for each box
            if (price > maxPrice[box]) {
                maxPrice[box] = price;
            }
        }

        int total = 0;
        for (int i = 1; i <= m; i++) {
            total += maxPrice[i]; // sum of chosen toys
        }

        System.out.println(total);
        sc.close();
    }
}
