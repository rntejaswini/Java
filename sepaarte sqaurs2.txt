class Solution {
    public double separateSquares(int[][] squares) {
        double totalArea = 0.0;
        double low = Double.MAX_VALUE;
        double high = Double.MIN_VALUE;

        // Calculate total area and search boundaries
        for (int[] sq : squares) {
            double y = sq[1];
            double l = sq[2];
            totalArea += l * l;
            low = Math.min(low, y);
            high = Math.max(high, y + l);
        }

        double target = totalArea / 2.0;

        // Binary search for y
        for (int i = 0; i < 100; i++) { // sufficient for 1e-5 precision
            double mid = (low + high) / 2.0;
            double areaBelow = areaBelow(mid, squares);

            if (areaBelow < target) {
                low = mid;
            } else {
                high = mid;
            }
        }

        return low;
    }

    private double areaBelow(double yLine, int[][] squares) {
        double area = 0.0;

        for (int[] sq : squares) {
            double bottom = sq[1];
            double top = bottom + sq[2];
            double side = sq[2];

            if (yLine <= bottom) {
                continue;
            } else if (yLine >= top) {
                area += side * side;
            } else {
                area += (yLine - bottom) * side;
            }
        }

        return area;
    }
}
