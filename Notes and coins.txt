import java.util.*;

abstract class Money {
    protected int val;

    public void setValue(int val) {
        this.val = val;
    }

    public int getValue() {
        return val;
    }
}

class Coin extends Money {
    // inherits setValue() and getValue()
}

class Note extends Money {
    // inherits setValue() and getValue()
}

class Bag<T extends Money> {
    private List<T> items = new ArrayList<>();

    public void add(T t) {
            items.add(t);
    }

    public void display() {
        for (T item : items) {
            System.out.println(item.getValue());
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        Bag<Coin> bagOfCoins = new Bag<>();
        Bag<Note> bagOfNotes = new Bag<>();

        for (int i = 0; i < n; i++) {
            String type = sc.next();
            int value = sc.nextInt();

            if (type.equals("Coin")) {
                Coin c = new Coin();
                c.setValue(value);
                bagOfCoins.add(c);
            } else if (type.equals("Note")) {
                Note no = new Note();
                no.setValue(value);
                bagOfNotes.add(no);
            }
        }

        System.out.println("Coins :");
        bagOfCoins.display();

        System.out.println("Notes :");
        bagOfNotes.display();
    }
}
