class Solution {
    public long maxMatrixSum(int[][] matrix) {
        long sum = 0;
        int negCount = 0;
        boolean hasZero = false;
        int minAbs = Integer.MAX_VALUE;

        for (int[] row : matrix) {
            for (int val : row) {
                if (val < 0) negCount++;
                if (val == 0) hasZero = true;

                int absVal = Math.abs(val);
                sum += absVal;
                minAbs = Math.min(minAbs, absVal);
            }
        }

        // If negatives are odd and no zero exists,
        // one smallest absolute element must remain negative
        if (negCount % 2 == 1 && !hasZero) {
            sum -= 2L * minAbs;
        }

        return sum;
    }
}
