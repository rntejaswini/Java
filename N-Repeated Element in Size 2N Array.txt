class Solution {
    public int repeatedNTimes(int[] nums) {
        java.util.HashSet<Integer> seen = new java.util.HashSet<>();
        
        for (int num : nums) {
            if (seen.contains(num)) {
                return num; // first repeated element
            }
            seen.add(num);
        }
        
        return -1; // this line theoretically never executes
    }
}
