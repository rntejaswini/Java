import java.util.TreeSet;

class Solution {
    public boolean containsNearbyAlmostDuplicate(int[] nums, int indexDiff, int valueDiff) {
        if (nums == null || nums.length < 2 || indexDiff <= 0) return false;
        
        TreeSet<Long> set = new TreeSet<>();
        
        for (int i = 0; i < nums.length; i++) {
            long num = (long) nums[i];
            
            // Check if there is a number in the set within valueDiff
            Long ceil = set.ceiling(num - valueDiff);
            if (ceil != null && ceil <= num + valueDiff) {
                return true;
            }
            
            set.add(num);
            
            // Maintain the sliding window size
            if (i >= indexDiff) {
                set.remove((long) nums[i - indexDiff]);
            }
        }
        
        return false;
    }
}
