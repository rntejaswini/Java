import java.io.*;
import java.util.*;

class TestClass {

    static class House {
        long v;   // y - x
        long people;

        House(long v, long p) {
            this.v = v;
            this.people = p;
        }
    }

    public static void main(String[] args) throws Exception {

        FastScanner fs = new FastScanner(System.in);
        StringBuilder out = new StringBuilder();

        int t = fs.nextInt();

        while (t-- > 0) {
            int n = fs.nextInt();

            House[] arr = new House[n];
            long total = 0;

            for (int i = 0; i < n; i++) {
                long x = fs.nextLong();
                long y = fs.nextLong();
                long p = fs.nextLong();

                long v = y - x;      // value relative to line y = x + c
                arr[i] = new House(v, p);
                total += p;
            }

            Arrays.sort(arr, (a, b) -> Long.compare(a.v, b.v));

            long leftSum = 0;
            boolean possible = false;

            int i = 0;
            while (i < n) {

                int j = i;
                long groupSum = 0;

                // group houses having same (y - x)
                while (j < n && arr[j].v == arr[i].v) {
                    groupSum += arr[j].people;
                    j++;
                }

                // Option 1: put line THROUGH this group (ignore them)
                long rightSum = total - leftSum - groupSum;
                if (leftSum == rightSum) {
                    possible = true;
                    break;
                }

                // add this group to left side
                leftSum += groupSum;

                // Option 2: put line BETWEEN this group and next group
                rightSum = total - leftSum;
                if (leftSum == rightSum) {
                    possible = true;
                    break;
                }

                i = j;
            }

            out.append(possible ? "YES\n" : "NO\n");
        }

        System.out.print(out.toString());
    }

    // Fast input reader
    static class FastScanner {
        private final byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;
        private final InputStream in;

        FastScanner(InputStream is) {
            in = is;
        }

        private int read() throws IOException {
            if (ptr >= len) {
                len = in.read(buffer);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        long nextLong() throws IOException {
            int c; long sign = 1, val = 0;
            do { c = read(); } while (c <= ' ' && c >= 0);
            if (c == '-') { sign = -1; c = read(); }
            for (; c > ' '; c = read())
                val = val * 10 + (c - '0');
            return val * sign;
        }

        int nextInt() throws IOException {
            return (int) nextLong();
        }
    }
}
