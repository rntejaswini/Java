class Solution {
    public ListNode deleteDuplicates(ListNode head) {

        ListNode curr = head;

        while (curr != null && curr.next != null) {

            if (curr.val == curr.next.val) {
                // Skip duplicate node
                curr.next = curr.next.next;
            } else {
                // Move to next distinct node
                curr = curr.next;
            }
        }

        return head;
    }
}
